    <script>
        // Menunggu semua elemen halaman dimuat sebelum menjalankan skrip
        document.addEventListener('DOMContentLoaded', () => {

            // Memilih elemen-elemen yang diperlukan dari HTML
            const form = document.getElementById('form-absensi');
            const tabelBody = document.getElementById('tabel-body');

            // Menambahkan event listener ke form saat tombol 'submit' ditekan
            form.addEventListener('submit', function(event) {
                // Mencegah form mengirim data dan me-refresh halaman
                event.preventDefault();

                // 1. Mengambil nilai (value) dari setiap input
                const namaLengkap = document.getElementById('nama-lengkap').value;
                const jenisKelamin = document.getElementById('jenis-kelamin').value;
                const kelas = document.getElementById('kelas').value;
                const nomorAbsen = document.getElementById('nomor-absen').value;

                // 2. Mendapatkan tanggal hari ini dengan format Indonesia
                const tanggalHariIni = new Date().toLocaleDateString('id-ID', {
                    year: 'numeric',
                    month: 'long',
                    day: 'numeric'
                });

                // 3. Membuat baris tabel baru (<tr>)
                const barisBaru = document.createElement('tr');

                // 4. Mengisi baris baru dengan data menggunakan innerHTML
                barisBaru.innerHTML = `
                    <td>${nomorAbsen}</td>
                    <td>${namaLengkap}</td>
                    <td>${jenisKelamin}</td>
                    <td>${kelas}</td>
                    <td class="keterangan-check"><input type="checkbox"></td>
                    <td>${tanggalHariIni}</td>
                    <td><button class="delete-btn">Hapus</button></td>
                `;

                // 5. Menambahkan baris baru ke dalam body tabel
                tabelBody.appendChild(barisBaru);

                // 6. Mengosongkan form setelah data berhasil ditambahkan
                form.reset();
            });

            // Menambahkan event listener untuk tombol hapus
            // Event listener ini diletakkan pada 'tabelBody' untuk menangani tombol hapus
            // yang baru dibuat secara dinamis (event delegation).
            tabelBody.addEventListener('click', function(event) {
                // Cek apakah elemen yang diklik memiliki class 'delete-btn'
                if (event.target.classList.contains('delete-btn')) {
                    // Menghapus elemen <tr> yang merupakan induk dari tombol yang diklik
                    const barisUntukDihapus = event.target.closest('tr');
                    barisUntukDihapus.remove();
                }
            });
        });
    </script>
